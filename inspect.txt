const dock=shadow.querySelector("#dock");
setTimeout(()=>dock.classList.add("open"),50);

shadow.getElementById("close").onclick=()=>{
dock.classList.remove("open");
setTimeout(()=>{
host.remove();
delete window.__aeroInspector;
},600);
};

const highlight=document.createElement("div");
highlight.className="highlight-box";
document.body.appendChild(highlight);
highlight.style.display="none";

let selected=null;

function updateSelectorList(){
const select=shadow.getElementById("elementList");
select.innerHTML="";
document.querySelectorAll("*").forEach(el=>{
const opt=document.createElement("option");
opt.value=el.tagName;
opt.textContent=el.tagName.toLowerCase() + (el.id?"#"+el.id:"");
select.appendChild(opt);
});
}

shadow.getElementById("pick").onclick=()=>{
document.body.style.cursor="crosshair";

function handler(e){
e.preventDefault();
e.stopPropagation();
selected=e.target;
document.body.style.cursor="default";
document.removeEventListener("click",handler,true);

const rect=selected.getBoundingClientRect();
highlight.style.display="block";
highlight.style.top=rect.top+"px";
highlight.style.left=rect.left+"px";
highlight.style.width=rect.width+"px";
highlight.style.height=rect.height+"px";

shadow.getElementById("styleEditor").value=selected.getAttribute("style")||"";
}

document.addEventListener("click",handler,true);
};

shadow.getElementById("apply").onclick=()=>{
if(selected){
selected.setAttribute("style",shadow.getElementById("styleEditor").value);
}
};

updateSelectorList();

})();
